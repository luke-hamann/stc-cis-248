Table TeamMembers {
  id int [ pk ]
  firstName string [ not null ]
  middleName string
  lastName string [ not null ]
  birthDate date [ not null ]
  email string
  phone string
  isExternal bool [ not null ]
  maxWeeklyHours double
  maxWeeklyDays int
  username string
  password string
  isAdmin bool [ not null ]
}

Table TeamMemberAvailability {
  id int [ pk ]
  teamMemberId int [ not null, ref: > TeamMembers.id ]
  startDateTime datetime [ not null ]
  endDateTime datetime [ not null ]
  isPreference bool [ not null ]
}

Table TeamMememberTypicalAvailability {
  id int [ pk ]
  teamMember int [ not null, ref: > TeamMembers.id ]
  dayOfWeek int [ not null ]
  startTime time [ not null ]
  endTime time [ not null ]
}

Table Substitutes {
  teamMemberId int [ pk, ref: > TeamMembers.id ]
  date date [ pk ]
}

Table TeamMemberShiftContextPreferences {
  teamMemberId int [ pk, ref: > TeamMembers.id ]
  shiftContextId int [ pk, ref: > ShiftContexts.id ]
  isPreferable bool [ not null ]
}

Table ShiftContexts {
  id int [ pk ]
  name string [ not null ]
  ageGroup string
  location string
  description string
}

Table TimeSlots {
  id int [ pk ]
  shiftContextId int [ not null, ref: > ShiftContexts.id ]
  startDateTime datetime [ not null ]
  endDateTime datetime
  requiresAdult bool [ not null ]
  teamMemberId int [ ref: > TeamMembers.id ]
  note string
  colorId int [ ref: > Colors.id ]
}

Table ShiftContextNotes {
  shiftContextId int [ pk, ref: > ShiftContexts.id ]
  date date [ pk ]
  note sting [ not null ]
  colorId int [ ref: > Colors.id ]
}

Table Colors {
  id int [ pk ]
  name string [ not null ]
  hex string [ not null ]
}
